def merge_sort(array, inversions):
    array_n = len(array)
    if array_n == 1:
        return array, inversions
    else:
        left = array[:array_n//2]
        right = array[array_n//2:]
 
        left, inversions = merge_sort(left, inversions)
        right, inversions = merge_sort(right, inversions)
 
    return merge(left, right, inversions)
 
 
def merge(left, right, inversions):
    res = []
    while len(left) > 0 and len(right) > 0:
        if left[0] > right[0]:
            res.append(right.pop(0))
            inversions += len(left)
        else:
            res.append(left.pop(0))
 
    if len(left) > 0:
        res = res+left
    if len(right) > 0:
        res = res+right
 
    return res, inversions
 
 
def count_inversions(array):
    inversions = 0
    return merge_sort(array, inversions)[1]
